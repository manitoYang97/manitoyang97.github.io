  • 博客内容展示

项目初始化

1.建立文件所需文件夹

  • public静态资源
  • model数据库操作
  • route路由
  • views模板

2.初始化项目描述文件

  • npm init -y

3.下载项目所需第三方模块

  • npm install express mongooser art-template express-art-template

4.创建网站服务器

//根目录的app.js文件

//引入Express框架
const express = require('express');
//创建网站服务器
const app = express();

//监听端口
app.listen(80);

5.构建模块化路由

在route文件夹下建立两个页面有两个路由home展示页面和admin管理页面

//route文件夹下面的home.js

//引入Express框架
const express = require('express');
//创建博客展示页面路由
const home = express.Router();

//挂载二级路由 博客首页
home.get('/', require('./home/index'));

//将路由对象作为模块成员进行导出
module.exports = home;
//要使路由生效必须在app.js里面去引入，然后用use拦截匹配路径

//引入路由模块
const home = require('./route/home');
const admin = require('./route/admin');
//为路由匹配请求路径
app.use('/home', home);
app.use('/admin', admin);

6.构建博客管理页面模块

​ 6.1.把静态资源放在public文件夹中，再把静态页面开放出去

//导入处理路径模块
const path = require('path');

//开放静态资源文件
app.use(express.static(path.join(__dirname, 'public')));

​ 6.2.要再.htnl文件里做一些动态的操作，所有它不算静态页面。要将它移动到views文件夹里并且后缀改为.art,然后再路由里渲染模板

//为避免每次都要配置模板路径，要在app.js中做统一设置

//告诉express框架模板所在的位置
app.set('views', path.join(__dirname, 'views'));
//告诉express框架模块的默认后缀
app.set('view engine', 'art');
//当渲染后缀为art的模板时，所使用的模板引擎是什么
app.engine('art', require('express-art-template'));
//渲染登录页面的路由 route/admin.js
admin.get('/login', require('./admin/loginPage'));

//  route/admin/loginPage.js
module.exports = (req, res) => {
    res.render('admin/login');
}

​ 6.3.模板外链资源，相对路径改为绝对路径，其中/就代表绝对路径，再外链资源前加上/admin/或/home/就可以了

​ 6.4. 优化模板，将公共部分和骨架抽离出来

//公共部分位置  views/admin/common/header.art

//在一个文件内引入回来
{{include './common/header.art'}}
//骨架部分位置 views/admin/common/layout.art
//不相同的地方要预留坑位 以不同的名字区分
{{block 'link'}}{{/block}}
{{block 'main'}}{{/block}}
{{block 'script'}}{{/block}}

//在文件中引入回来 先继承再填坑
{{extend './common/layout.art'}}
{{block 'main'}}
<p class='bg-danger error'>{{msg}}</p>
{{/block}}

登录

  1. 创建用户集合，初始化用户

    ​ 1.1连接数据库（model/connect.js）

    //引入mongoose这个第三方模块
const mongoose = require('mongoose');

//连接数据库
mongoose.connect(`mongodb://${config.get('db.user')}:${config.get('db.pwd')}@${config.get('db.host')}:${config.get('db.port')}/${config.get('db.name')}`, { useNewUrlParser: true })
    .then(() => console.log('数据库连接成功'))
    .catch(() => console.log('数据库连接失败'))

//在app.js中引入
//数据库连接
require('./model/connect');

    ​ 1.2创建用户集合（model/user.js）

    //引入mongoose这个第三方模块
const mongoose = require('mongoose');

//创建集合规则
const userSchema = new mongoose.Schema({
    username: {
        type: String,
        required: true,
        minlength: 2,
        maxlength: 20
    },
    email: {
        type: String,
        //保证唯一性
        unique: true,
        required: true
    },
    password: {
        type: String,
        required: true
    },
    //admin超级管理员  normal普通用户
    role: {
        type: String,
        required: true
    },
    //状态0为启用 1为禁用
    state: {
        type: Number,
        default: 0
    }
});
//创建集合
const User = mongoose.model('User', userSchema);

//将用户集合作为模块成员进行导出
module.exports = {User}

    ​ 1.3 初始化用户

    //导入bcrypt加密模块
const bcrypt = require('bcrypt');

//创建文档
async function createUser() {
    const salt = await bcrypt.genSalt(10);
    const pass = await bcrypt.hash('123456', salt);
    const user = await User.create({
        username: 'yang',

  2. 为登录表单项设置请求地址、请求方式以及表单项name属性

  3. 当用户点击登录按钮时，客户端验证用户是否填写了登录表单

    // 用户点击登录按钮获取用户填写内容 public/admin/js/common.js

function serializeToJson(form) {
    var result = {};
    //serializeArray()获取表单用户输入的内容[{name:'',value:'用户输入的内容'}]
    var f = form.serializeArray();
    f.forEach(function(item) {
        result[item.name] = item.value;
    });
    //{email:'',password:''}
    return result;
}

  5. 服务器端接收请求参数，验证用户是否填写了登录表单

  6. 如果其中一项没有输入，为客户端做出响应，阻止程序向下执行

    ​ 先下载接收post请求参数的第三方模块npm install body-parser

admin.post('/login', require('./admin/login'))
    //实现登录功能  route/admin/login.js
module.exports = async(req, res) => {
   //接收请求参数
   const { email, password } = req.body;
   if (email.trim().length == 0 || password.trim().length == 0) {
       return res.status(400).render('admin/error', { msg: '邮件地址或密码错误' })
   }
